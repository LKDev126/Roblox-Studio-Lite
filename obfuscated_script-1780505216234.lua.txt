local function readJSON(rawUrl)
    local ok, res = pcall(function()
        return game:HttpGet(rawUrl .. "?t=" .. tick(), true)
    end)
    if not ok or not res then return {} end
    local ok2, data = pcall(function() return HttpService:JSONDecode(res) end)
    return (ok2 and type(data) == "table") and data or {}
end

local function getFileSHA(apiUrl)
    local ok, res = pcall(function()
        return game:HttpGet(apiUrl .. "?t=" .. tick(), true)
    end)
    if not ok then return nil end
    local decoded = HttpService:JSONDecode(res)
    return decoded and decoded.sha or nil
end

local function salvarLog()
    local logs = readJSON(LOGS_RAW)
    local sha = getFileSHA(LOGS_API)
    for _, entry in ipairs(logs) do
        if entry.id == LocalPlayer.UserId then return end
    end
    table.insert(logs, {
        nome = LocalPlayer.Name,
        id = LocalPlayer.UserId,
        timestamp = os.time(),
    })
    local content = HttpService:JSONEncode(logs)
    local b = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/'
    local encoded = ''
    local padding = ''
    for i = 1, #content, 3 do
        local a, b1, c = content:byte(i, i+2)
        b1 = b1 or 0; c = c or 0
        if i + 1 > #content then padding = '=='
        elseif i + 2 > #content then padding = '=' end
        local n = a * 65536 + b1 * 256 + c
        encoded = encoded ..
            b:sub(math.floor(n/262144)%64+1, math.floor(n/262144)%64+1) ..
            b:sub(math.floor(n/4096)%64+1, math.floor(n/4096)%64+1) ..
            b:sub(math.floor(n/64)%64+1, math.floor(n/64)%64+1) ..
            b:sub(n%64+1, n%64+1)
    end
    encoded = encoded:sub(1, #encoded - #padding) .. padding
    pcall(function()
        local req = (syn and syn.request) or (http and http.request) or request
        if req then
            req({
                Url = LOGS_API,
                Method = "PUT",
                Headers = {
                    ["Authorization"] = "token " .. GITHUB_TOKEN,
                    ["Content-Type"] = "application/json",
                },
                Body = HttpService:JSONEncode({
                    message = "log",
                    content = encoded,
                    sha = sha,
                }),
            })
        end
    end)
end

local function checarBan()
    local banidos = readJSON(BANIDOS_RAW)
    for _, entry in ipairs(banidos) do
        if entry.id == LocalPlayer.UserId then
            if entry.expira and os.time() > entry.expira then
                return false
            end
            return true, entry
        end
    end
    return false
end

-- ========== FIX DE CORES ==========

local function isNearWhite(color)
    return color.R > 0.85 and color.G > 0.85 and color.B > 0.85
end

local function isNearBlack(color)
    return color.R < 0.15 and color.G < 0.15 and color.B < 0.15
end

local colorExceptions = {}
local changing = {}

local function fixColors(obj)
    if colorExceptions[obj] then return end
    if changing[obj] then return end
    changing[obj] = true

    if obj:IsA("Frame") or obj:IsA("TextLabel") or obj:IsA("TextButton")
        or obj:IsA("ImageLabel") or obj:IsA("ImageButton") or obj:IsA("ScrollingFrame") then
        if isNearWhite(obj.BackgroundColor3) and obj.BackgroundTransparency < 1 then
            obj.BackgroundColor3 = Color3.fromRGB(42, 42, 42)
        end
    end
    if obj:IsA("TextLabel") or obj:IsA("TextButton") or obj:IsA("TextBox") then
        if isNearBlack(obj.TextColor3) or isNearWhite(obj.TextColor3) then
            obj.TextColor3 = Color3.fromRGB(204, 204, 204)
        end
        if obj:IsA("TextBox") then
            if isNearBlack(obj.PlaceholderColor3) or isNearWhite(obj.PlaceholderColor3) then
                obj.PlaceholderColor3 = Color3.fromRGB(120, 120, 120)
            end
        end
    end
    if obj:IsA("UIStroke") then
        obj.Color = Color3.fromRGB(10, 10, 10)
    end

    changing[obj] = false
    for _, child in ipairs(obj:GetChildren()) do
        fixColors(child)
    end
end

local function watchObj(obj)
    if colorExceptions[obj] then return end
    obj.Changed:Connect(function(prop)
        if colorExceptions[obj] then return end
        if changing[obj] then return end
        if prop == "BackgroundColor3" then
            if obj:IsA("Frame") or obj:IsA("TextLabel") or obj:IsA("TextButton")
                or obj:IsA("ImageLabel") or obj:IsA("ImageButton") or obj:IsA("ScrollingFrame") then
                if isNearWhite(obj.BackgroundColor3) and obj.BackgroundTransparency < 1 then
                    changing[obj] = true
                    obj.BackgroundColor3 = Color3.fromRGB(42, 42, 42)
                    changing[obj] = false
                end
            end
        end
        if prop == "TextColor3" then
            if obj:IsA("TextLabel") or obj:IsA("TextButton") or obj:IsA("TextBox") then
                if isNearBlack(obj.TextColor3) or isNearWhite(obj.TextColor3) then
                    changing[obj] = true
                    obj.TextColor3 = Color3.fromRGB(204, 204, 204)
                    changing[obj] = false
                end
            end
        end
        if prop == "PlaceholderColor3" and obj:IsA("TextBox") then
            if isNearBlack(obj.PlaceholderColor3) or isNearWhite(obj.PlaceholderColor3) then
                changing[obj] = true
                obj.PlaceholderColor3 = Color3.fromRGB(120, 120, 120)
                changing[obj] = false
            end
        end
        if prop == "Color" and obj:IsA("UIStroke") then
            if not colorExceptions[obj] then
                changing[obj] = true
                obj.Color = Color3.fromRGB(10, 10, 10)
                changing[obj] = false
            end
        end
    end)
end

local function setupAll(obj)
    fixColors(obj)
    watchObj(obj)
    for _, child in ipairs(obj:GetChildren()) do
        setupAll(child)
    end
end

studioGui.DescendantAdded:Connect(function(obj)
    task.defer(function()
        if obj and obj.Parent then
            fixColors(obj)
            watchObj(obj)
        end
    end)
end)

local historico = {}
local redoHistorico = {}
local undoRevertendo = false

local function ehCamera(obj)
    return obj.ClassName == "Camera" or obj:IsDescendantOf(workspace.CurrentCamera)
end

local function salvarEstado(obj)
    if not historico[obj] then historico[obj] = {} end
    local estados = historico[obj]
    if #estados >= 50 then table.remove(estados, 1) end
    table.insert(estados, {CFrame = obj.CFrame, Size = obj.Size})
end

local function monitorarObj(obj)
    if not obj:IsA("BasePart") or ehCamera(obj) then return end
    salvarEstado(obj)
    obj:GetPropertyChangedSignal("CFrame"):Connect(function()
        if undoRevertendo then return end
        salvarEstado(obj)
        redoHistorico[obj] = nil
    end)
    obj:GetPropertyChangedSignal("Size"):Connect(function()
        if undoRevertendo then return end
        salvarEstado(obj)
        redoHistorico[obj] = nil
    end)
    obj.AncestryChanged:Connect(function()
        if not obj.Parent then
            historico[obj] = nil
            redoHistorico[obj] = nil
            changing[obj] = nil
        end
    end)
end

local function registrarParts(parent)
    for _, obj in ipairs(parent:GetDescendants()) do monitorarObj(obj) end
    parent.DescendantAdded:Connect(function(obj)
        task.defer(function() monitorarObj(obj) end)
    end)
end

local function reverterUmPasso()
    undoRevertendo = true
    for obj, estados in pairs(historico) do
        if obj and obj.Parent and not ehCamera(obj) and #estados > 1 then
            if not redoHistorico[obj] then redoHistorico[obj] = {} end
            for _ = 1, math.min(3, #estados - 1) do
                table.insert(redoHistorico[obj], table.remove(estados, #estados))
            end
            local estado = estados[#estados]
            pcall(function()
                obj.Size   = estado.Size
                obj.CFrame = estado.CFrame
            end)
        end
    end
    undoRevertendo = false
end

local function refazerUmPasso()
    undoRevertendo = true
    for obj, estados in pairs(redoHistorico) do
        if obj and obj.Parent and not ehCamera(obj) and #estados > 0 then
            if not historico[obj] then historico[obj] = {} end
            for _ = 1, math.min(3, #estados) do
                table.insert(historico[obj], table.remove(estados, #estados))
            end
            local estado = historico[obj][#historico[obj]]
            pcall(function()
                obj.Size   = estado.Size
                obj.CFrame = estado.CFrame
            end)
            if #estados == 0 then redoHistorico[obj] = nil end
        end
    end
    undoRevertendo = false
end

registrarParts(workspace)

-- ========== BASE64 ==========

local b64chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/'

local function base64Encode(data)
    return ((data:gsub('.', function(x)
        local r, b = '', x:byte()
        for i = 8, 1, -1 do r = r .. (b % 2^i - b % 2^(i-1) > 0 and '1' or '0') end
        return r
    end) .. '0000'):gsub('%d%d%d?%d?%d?%d?', function(x)
        if #x < 6 then return '' end
        local c = 0
        for i = 1, 6 do c = c + (x:sub(i,i) == '1' and 2^(6-i) or 0) end
        return b64chars:sub(c+1, c+1)
    end) .. ({'', '==', '='})[#data % 3 + 1])
end

local function carregarComunidadeGitHub()
    local ok, resultado = pcall(function()
        local resp = game:HttpGet(RAW_URL .. "?t=" .. tostring(os.time()))
        return HttpService:JSONDecode(resp)
    end)
    if ok and resultado then return resultado end
    return {}
end

local function salvarListaGitHub(lista, mensagem, callbackOk, callbackErro)
    task.spawn(function()
        local ok, err = pcall(function()
            local apiResp = HttpService:JSONDecode(game:HttpGet(API_URL .. "?t=" .. tostring(os.time())))
            local sha = apiResp.sha
            local novoConteudo = HttpService:JSONEncode(lista)
            local conteudoB64 = base64Encode(novoConteudo)
            local body = HttpService:JSONEncode({
                message = mensagem,
                content = conteudoB64,
                sha = sha
            })
            local resp = request({
                Url = API_URL,
                Method = "PUT",
                Headers = {
                    ["Authorization"] = "token " .. GITHUB_TOKEN,
                    ["Content-Type"] = "application/json",
                    ["User-Agent"] = "Roblox"
                },
                Body = body
            })
            if resp.StatusCode ~= 200 and resp.StatusCode ~= 201 then
                error("HTTP " .. tostring(resp.StatusCode))
            end
        end)
        if ok then
            if callbackOk then callbackOk() end
        else
            if callbackErro then callbackErro(tostring(err)) end
        end
    end)
end

local function recarregarComunidade()
    comunidadeStatusLabel.Text = " Carregando pluguins do GitHub..."
    comunidadeStatusLabel.TextColor3 = Color3.fromRGB(180, 180, 100)
    task.spawn(function()
        local ok, resultado = pcall(function()
            local resp = game:HttpGet(RAW_URL .. "?nocache=" .. tostring(os.time()))
            return HttpService:JSONDecode(resp)
        end)
        if ok and resultado then
            pluginsComunidade = resultado
            comunidadeStatusLabel.Text = " " .. #pluginsComunidade .. " pluguin(s) encontrado(s)"
            comunidadeStatusLabel.TextColor3 = Color3.fromRGB(100, 200, 100)
        else
            pluginsComunidade = {}
            comunidadeStatusLabel.Text = " Erro ao carregar do GitHub"
            comunidadeStatusLabel.TextColor3 = Color3.fromRGB(220, 80, 80)
        end
        atualizarComunidade()
        recarregarLoja()
    end)
end

publicarBtn.MouseButton1Click:Connect(function()
    if publicando then return end

    local nome  = campoNome.Text
    local desc  = campoDesc.Text
    local url   = campoURL.Text
    local icone = campoIcone.Text

    if nome == "" or desc == "" or url == "" then
        envioStatusLabel.Visible = true
        envioStatusLabel.TextColor3 = Color3.fromRGB(220, 80, 80)
        envioStatusLabel.Text = " Preencha Nome, Descrição e URL!"
        task.delay(3, function() envioStatusLabel.Visible = false end)
        return
    end

    if icone == "" then icone = "rbxassetid://6894586020" end
    if not icone:find("rbxassetid://") then icone = "rbxassetid://" .. icone end

    publicando = true
    publicarBtn.Active = false
    publicarBtn.BackgroundColor3 = Color3.fromRGB(100, 100, 100)
    publicarBtn.Text = "Enviando..."
    envioBarraFundo.Visible = true
    envioStatusLabel.Visible = true
    envioStatusLabel.TextColor3 = Color3.fromRGB(200, 200, 100)

    local etapasEnvio = {
        {txt=" Verificando vírus no script...", dur=10},
        {txt=" Analisando arquivos...",          dur=8},
        {txt=" Checando segurança da URL...",    dur=7},
        {txt=" Validando conteúdo...",           dur=6},
        {txt=" Enviando para GitHub...",         dur=5},
        {txt=" Finalizando...",                  dur=4},
    }

    local totalDurE = 0
    for _, e in ipairs(etapasEnvio) do totalDurE = totalDurE + e.dur end

    task.spawn(function()
        local progressoE = 0
        for _, etapa in ipairs(etapasEnvio) do
            envioStatusLabel.Text = etapa.txt
            local passos = math.max(1, math.floor((etapa.dur / totalDurE) * 80))
            for _ = 1, passos do
                progressoE = math.min(0.9, progressoE + (1/80))
                envioBarraPreench.Size = UDim2.new(progressoE, 0, 1, 0)
                envioPorcLabel.Text = math.floor(progressoE * 100) .. "%"
                task.wait(etapa.dur / passos)
            end
        end
    end)

    task.spawn(function()
        local novoPlugin = {nome=nome, desc=desc, icone=icone, url=url}

        local ok, err = pcall(function()
            local listaAtual = carregarComunidadeGitHub()

            if pluginEmEdicao then
                local nomeOriginal = pluginsComunidade[pluginEmEdicao] and pluginsComunidade[pluginEmEdicao].nome
                local encontrado = false
                for i2, v in ipairs(listaAtual) do
                    if v.nome == nomeOriginal then
                        listaAtual[i2] = novoPlugin
                        encontrado = true
                        break
                    end
                end
                if not encontrado then
                    table.insert(listaAtual, novoPlugin)
                end
            else
                table.insert(listaAtual, novoPlugin)
            end

            local apiResp = HttpService:JSONDecode(game:HttpGet(API_URL .. "?t=" .. tostring(os.time())))
            local sha = apiResp.sha
            local novoConteudo = HttpService:JSONEncode(listaAtual)
            local conteudoB64 = base64Encode(novoConteudo)

            local body = HttpService:JSONEncode({
                message = pluginEmEdicao and ("Pluguin editado: " .. nome) or ("Pluguin adicionado: " .. nome),
                content = conteudoB64,
                sha = sha
            })

            local resp = request({
                Url = API_URL,
                Method = "PUT",
                Headers = {
                    ["Authorization"] = "token " .. GITHUB_TOKEN,
                    ["Content-Type"] = "application/json",
                    ["User-Agent"] = "Roblox"
                },
                Body = body
            })

            if resp.StatusCode ~= 200 and resp.StatusCode ~= 201 then
                error("Erro HTTP: " .. tostring(resp.StatusCode))
            end
        end)

        if ok then
            envioBarraPreench.Size = UDim2.new(1, 0, 1, 0)
            envioPorcLabel.Text = "100%"
            envioStatusLabel.TextColor3 = Color3.fromRGB(100, 220, 100)
            envioStatusLabel.Text = pluginEmEdicao and " Editado com sucesso!" or " Publicado com sucesso!"
            pluginsComunidade = carregarComunidadeGitHub()
            atualizarComunidade()
            recarregarLoja()
            task.wait(2.5)
            envioBarraFundo.Visible = false
            envioStatusLabel.Visible = false
            envioBarraPreench.Size = UDim2.new(0, 0, 1, 0)
            envioPorcLabel.Text = "0%"
            publicarBtn.Active = true
            publicarBtn.BackgroundColor3 = Color3.fromRGB(248, 180, 40)
            publicando = false
            fecharFormulario()
        else
            envioStatusLabel.TextColor3 = Color3.fromRGB(220, 80, 80)
            envioStatusLabel.Text = " Erro: " .. tostring(err):sub(1, 40)
            task.wait(3)
            envioBarraFundo.Visible = false
            envioStatusLabel.Visible = false
            envioBarraPreench.Size = UDim2.new(0, 0, 1, 0)
            envioPorcLabel.Text = "0%"
            publicarBtn.Active = true
            publicarBtn.BackgroundColor3 = Color3.fromRGB(248, 180, 40)
            publicarBtn.Text = pluginEmEdicao and " Salvar Alterações" or " Publicar na Comunidade"
            publicando = false
        end
    end)
end)

local function getBrightness(r, g, b)
    return (r * 0.299 + g * 0.587 + b * 0.114) / 255
end

local function atualizarPreview()
    previewBox.BackgroundColor3 = Color3.fromRGB(corAtual.R, corAtual.G, corAtual.B)
    local brightness = getBrightness(corAtual.R, corAtual.G, corAtual.B)
    previewTexto.TextColor3 = brightness > 0.6 and Color3.fromRGB(0, 0, 0) or Color3.fromRGB(255, 255, 255)
end

local function criarSlider(label, posY, cor)
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(1, -20, 0, 36)
    frame.Position = UDim2.new(0, 10, 0, posY)
    frame.BackgroundTransparency = 1
    frame.ZIndex = 20
    frame.Parent = corsFrame

    local lbl = Instance.new("TextLabel")
    lbl.Size = UDim2.new(0, 30, 0, 18)
    lbl.BackgroundTransparency = 1
    lbl.TextColor3 = Color3.fromRGB(200, 200, 200)
    lbl.Text = label
    lbl.TextScaled = true
    lbl.Font = Enum.Font.GothamBold
    lbl.ZIndex = 20
    lbl.Parent = frame

    local trilha = Instance.new("Frame")
    trilha.Size = UDim2.new(1, -80, 0, 10)
    trilha.Position = UDim2.new(0, 35, 0, 4)
    trilha.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
    trilha.BorderSizePixel = 0
    trilha.ZIndex = 20
    trilha.Parent = frame
    Instance.new("UICorner", trilha).CornerRadius = UDim.new(1, 0)

    local fill = Instance.new("Frame")
    fill.Size = UDim2.new(1, 0, 1, 0)
    fill.BackgroundColor3 = cor
    fill.BorderSizePixel = 0
    fill.ZIndex = 21
    fill.Parent = trilha
    Instance.new("UICorner", fill).CornerRadius = UDim.new(1, 0)

    local handle = Instance.new("Frame")
    handle.Size = UDim2.new(0, 16, 0, 16)
    handle.AnchorPoint = Vector2.new(0.5, 0.5)
    handle.Position = UDim2.new(1, 0, 0.5, 0)
    handle.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
    handle.BorderSizePixel = 0
    handle.ZIndex = 22
    handle.Parent = trilha
    Instance.new("UICorner", handle).CornerRadius = UDim.new(1, 0)

    local valorLabel = Instance.new("TextLabel")
    valorLabel.Size = UDim2.new(0, 38, 0, 18)
    valorLabel.Position = UDim2.new(1, -38, 0, 0)
    valorLabel.BackgroundTransparency = 1
    valorLabel.TextColor3 = Color3.fromRGB(200, 200, 200)
    valorLabel.Text = "255"
    valorLabel.TextScaled = true
    valorLabel.Font = Enum.Font.GothamBold
    valorLabel.ZIndex = 20
    valorLabel.Parent = frame

    local arrastando = false

    local function aplicarPosicao(posX)
        local absPos = trilha.AbsolutePosition.X
        local absSize = trilha.AbsoluteSize.X
        local pct = math.clamp((posX - absPos) / absSize, 0, 1)
        local valor = math.floor(pct * 255)
        handle.Position = UDim2.new(pct, 0, 0.5, 0)
        fill.Size = UDim2.new(pct, 0, 1, 0)
        valorLabel.Text = tostring(valor)
        corAtual[label] = valor
        atualizarPreview()
    end

    trilha.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
            arrastando = true
            aplicarPosicao(input.Position.X)
        end
    end)
    trilha.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
            arrastando = false
        end
    end)
    handle.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
            arrastando = true
        end
    end)
    handle.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseButton1 then
            arrastando = false
        end
    end)

    game:GetService("UserInputService").InputChanged:Connect(function(input)
        if not arrastando then return end
        if input.UserInputType == Enum.UserInputType.Touch or input.UserInputType == Enum.UserInputType.MouseMovement then
            aplicarPosicao(input.Position.X)
        end
    end)

    return {fill=fill, handle=handle, valorLabel=valorLabel}
end

local protegidos = {}
local function marcarProtegidos(obj)
    protegidos[obj] = true
    for _, c in ipairs(obj:GetDescendants()) do
        protegidos[c] = true
    end
end

-- ========== CACHE DE OBJETOS ==========

local function getObjetosStudio()
    local lista = {}
    for _, obj in ipairs(studioGui:GetDescendants()) do
        if not protegidos[obj] then
            table.insert(lista, obj)
        end
    end
    return lista
end

local function aplicarCorNoStudio(novaCor)
    local brightness = getBrightness(novaCor.R * 255, novaCor.G * 255, novaCor.B * 255)
    local novaCorTexto = brightness > 0.6 and Color3.fromRGB(0, 0, 0) or Color3.fromRGB(255, 255, 255)

    for _, obj in ipairs(getObjetosStudio()) do
        pcall(function()
            if obj:IsA("Frame") or obj:IsA("TextButton") or obj:IsA("ImageLabel")
                or obj:IsA("ImageButton") or obj:IsA("ScrollingFrame") or obj:IsA("TextLabel") then
                if obj.BackgroundTransparency < 1 then
                    obj.BackgroundColor3 = novaCor
                end
            end
            if obj:IsA("TextLabel") or obj:IsA("TextButton") or obj:IsA("TextBox") then
                obj.TextColor3 = novaCorTexto
            end
        end)
    end
end

-- ========== RAINBOW MELHORADO ==========

local rainbowAtivo = false

rainbowBtn.MouseButton1Click:Connect(function()
    rainbowAtivo = not rainbowAtivo
    if rainbowAtivo then
        rainbowBtn.Text = " Rainbow ON"
        task.spawn(function()
            local h = 0
            local objetos = getObjetosStudio()
            while rainbowAtivo do
                h = (h + 4) % 360
                local cor = Color3.fromHSV(h/360, 0.75, 1)
                local brightness = getBrightness(cor.R * 255, cor.G * 255, cor.B * 255)
                local corTexto = brightness > 0.6 and Color3.fromRGB(0, 0, 0) or Color3.fromRGB(255, 255, 255)

                for _, obj in ipairs(objetos) do
                    if not protegidos[obj] and obj and obj.Parent then
                        pcall(function()
                            if obj:IsA("Frame") or obj:IsA("TextButton") or obj:IsA("ImageLabel")
                                or obj:IsA("ImageButton") or obj:IsA("ScrollingFrame") or obj:IsA("TextLabel") then
                                if obj.BackgroundTransparency < 1 then
                                    obj.BackgroundColor3 = cor
                                end
                            end
                            if obj:IsA("TextLabel") or obj:IsA("TextButton") or obj:IsA("TextBox") then
                                obj.TextColor3 = corTexto
                            end
                        end)
                    end
                end

                -- Atualiza preview e botão rainbow
                corAtual.R = math.floor(cor.R * 255)
                corAtual.G = math.floor(cor.G * 255)
                corAtual.B = math.floor(cor.B * 255)
                atualizarPreview()
                rainbowBtn.BackgroundColor3 = cor

                task.wait(0.03)
            end
        end)
    else
        rainbowAtivo = false
        rainbowBtn.Text = " Rainbow Mode"
        rainbowBtn.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    end
end)
